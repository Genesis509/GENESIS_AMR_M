import cv2
from rclpy.node import Node
from std_msgs.msg import String
import rclpy
from rclpy.executors import MultiThreadedExecutor

class CameraSubscriber(Node):
    def __init__(self):
        super().__init__('camera_subscriber')
        self.cap = None
        
        self.sub = self.create_subscription(
            String, 'stream_info', self.on_stream_url, 10)
        self.timer = self.create_timer(1.0/20, self.show_frame)

    def on_stream_url(self, msg):
        if self.cap:
            return
        
        url = msg.data
        self.get_logger().info(f'Connecting to: {url}')
        
        gst_pipeline = (
            f'rtspsrc location={url} protocols=tcp latency=0 buffer-mode=0 ! '
            'rtph264depay ! h264parse config-interval=-1 ! avdec_h264 lowres=0 ! '
            'queue max-size-buffers=1 leaky=downstream ! '
            'videoconvert ! appsink drop=1 max-buffers=1 sync=false'
        )
        
        self.cap = cv2.VideoCapture(gst_pipeline, cv2.CAP_GSTREAMER)
        
        if not self.cap.isOpened():
            self.get_logger().error('Failed to open GStreamer pipeline')
            self.cap = None  # Reset so it can retry

    def show_frame(self):
        if not self.cap:
            return
        ret, frame = self.cap.read()
        if ret:
            cv2.imshow('Robot Camera', frame)
            cv2.waitKey(1)

def main(args=None):
    """Main entry point."""
    rclpy.init(args=args)
    node = None

    try:
        node = CameraSubscriber()
        rclpy.spin(node)

    except KeyboardInterrupt:
        pass

    except Exception as e:
        print(f"Error: {e}")

    finally:
        if node is not None:
            node.destroy_node()
        rclpy.shutdown()
        cv2.destroyAllWindows()


if __name__ == '__main__':
    main()
