#!/usr/bin/env python3
"""
ROS2 Camera Publisher Node for Raspberry Pi
Publishes compressed images directly for efficient streaming over network.

Author: Genesis
"""

import rclpy
from rclpy.node import Node
from rclpy.qos import QoSProfile, ReliabilityPolicy, HistoryPolicy, DurabilityPolicy
from sensor_msgs.msg import CompressedImage
from std_msgs.msg import Header
from picamera2 import Picamera2
import cv2
import numpy as np
from typing import Optional


class CameraPublisher(Node):
    """Publishes compressed camera images from Raspberry Pi camera."""

    def __init__(self):
        super().__init__('camera_publisher')

        # Declare parameters with defaults
        self.declare_parameter('frame_rate', 10.0)
        self.declare_parameter('width', 640)
        self.declare_parameter('height', 480)
        self.declare_parameter('jpeg_quality', 80)
        self.declare_parameter('topic_name', 'camera/image_raw/compressed')

        # Get parameters
        self.frame_rate = self.get_parameter('frame_rate').value
        self.width = self.get_parameter('width').value
        self.height = self.get_parameter('height').value
        self.jpeg_quality = self.get_parameter('jpeg_quality').value
        topic_name = self.get_parameter('topic_name').value

        # QoS profile optimized for real-time video streaming
        # - Best effort: drops frames rather than queuing (reduces latency)
        # - Volatile: no need to store for late subscribers
        # - Keep last 1: only most recent frame matters
        qos_profile = QoSProfile(
            reliability=ReliabilityPolicy.BEST_EFFORT,
            durability=DurabilityPolicy.VOLATILE,
            history=HistoryPolicy.KEEP_LAST,
            depth=1
        )

        # Publisher for compressed images
        self.image_pub = self.create_publisher(
            CompressedImage,
            topic_name,
            qos_profile
        )

        # Initialize camera
        self.picam2: Optional[Picamera2] = None
        self._init_camera()

        # Create capture timer
        timer_period = 1.0 / self.frame_rate
        self.timer = self.create_timer(timer_period, self.capture_and_publish)

        # Statistics
        self.frame_count = 0
        #self.stats_timer = self.create_timer(5.0, self._log_stats)

        self.get_logger().info(
            f"Camera Publisher started: {self.width}x{self.height} @ {self.frame_rate}fps, "
            f"JPEG quality: {self.jpeg_quality}"
        )

    def _init_camera(self) -> None:
        """Initialize the Picamera2 with configured settings."""
        try:
            self.picam2 = Picamera2()

            # Video configuration for streaming
            config = self.picam2.create_video_configuration(
                main={
                    'format': 'RGB888',
                    'size': (self.width, self.height)
                },
                buffer_count=4  # Buffer for smooth capture
            )
            self.picam2.configure(config)
            self.picam2.start()

            self.get_logger().info("Picamera2 initialized successfully")

        except Exception as e:
            self.get_logger().error(f"Failed to initialize camera: {e}")
            raise RuntimeError(f"Camera initialization failed: {e}")

    def capture_and_publish(self) -> None:
        """Capture frame from camera and publish as compressed image."""
        if self.picam2 is None:
            return

        try:
            # Capture frame (RGB format)
            frame_rgb = self.picam2.capture_array()

            # Convert RGB to BGR for OpenCV
            frame_bgr = cv2.cvtColor(frame_rgb, cv2.COLOR_RGB2BGR)

            # Compress to JPEG
            encode_params = [cv2.IMWRITE_JPEG_QUALITY, self.jpeg_quality]
            success, encoded_image = cv2.imencode('.jpg', frame_bgr, encode_params)

            if not success:
                self.get_logger().warning("Failed to encode frame")
                return

            # Create and publish message
            msg = CompressedImage()
            msg.header = Header()
            msg.header.stamp = self.get_clock().now().to_msg()
            msg.header.frame_id = 'camera_frame'
            msg.format = 'jpeg'
            msg.data = encoded_image.tobytes()

            self.image_pub.publish(msg)
            self.frame_count += 1

        except Exception as e:
            self.get_logger().error(f"Error capturing/publishing frame: {e}")

    def _log_stats(self) -> None:
        """Log publishing statistics."""
        fps = self.frame_count / 5.0
        self.get_logger().info(f"Publishing at {fps:.1f} fps")
        self.frame_count = 0

    def destroy_node(self) -> None:
        """Clean up resources."""
        self.get_logger().info("Shutting down camera publisher...")

        # Stop timer
        if hasattr(self, 'timer'):
            self.timer.cancel()

        # Stop and close camera
        if self.picam2 is not None:
            try:
                self.picam2.stop()
                self.picam2.close()
                self.get_logger().info("Camera stopped and closed")
            except Exception as e:
                self.get_logger().warning(f"Error closing camera: {e}")

        super().destroy_node()


def main(args=None):
    """Main entry point."""
    rclpy.init(args=args)
    node = None

    try:
        node = CameraPublisher()
        rclpy.spin(node)

    except KeyboardInterrupt:
        pass

    except Exception as e:
        print(f"Error: {e}")

    finally:
        if node is not None:
            node.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()
